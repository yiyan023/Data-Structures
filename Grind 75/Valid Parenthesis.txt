class Solution:
    def isValid(self, s: str) -> bool:
        dict = {
            ")" : "(",
            "]" : "[",
            "}" : "{"
        }

        stack = []

        for char in s:
            if char in dict:
                if stack and stack[-1] == dict[char]:
                    stack.pop()
                else:
                    return False

            else:
                stack.append(char)
        
        return len(stack) == 0
